async function uploadToImgbb(filePath) {
  try {
    if (!filePath) throw new Error('File path is required');
    if (!fs.existsSync(filePath)) throw new Error('File does not exist');

    const fileStream = fs.createReadStream(filePath);
    const res = await axios.get('https://imgbb.com');
    const auth_token = res.data.match(/auth_token="([^"]+)"/)[1];
    const timestamp = Date.now();

    const formData = new FormData();
    formData.append('source', fileStream);
    formData.append('type', 'file');
    formData.append('action', 'upload');
    formData.append('timestamp', timestamp);
    formData.append('auth_token', auth_token);

    const uploadRes = await axios.post('https://imgbb.com/json', formData, {
      headers: {
        'Content-Type': 'multipart/form-data'
      }
    });

    return uploadRes.data.image.url;
  } catch (error) {
    console.error('Upload error:', error);
    throw error;
  }
}

class Bard {
    static JSON = "json";
    static MD = "markdown";

    SNlM0e;
    headers;
    initPromise;
    #bardURL = "https://gemini.google.com";
    verbose = false;
    #fetch = fetch;

    constructor(cookie, config) {
        if (config?.verbose) this.verbose = true;
        if (config?.fetch) this.#fetch = config.fetch;

        if (!cookie) throw new Error("Cookie is required");
        this.cookie = cookie;
        this.initPromise = this.#init(cookie);
    }

    async #init(cookie) {
        this.verbose && console.log("Initializing Bard");

        this.headers = {
            Host: "bard.google.com",
            "X-Same-Domain": "1",
            "User-Agent": "Mozilla/5.0 (Windows NT 10.0; WOW64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/91.0.4472.114 Safari/537.36",
            "Content-Type": "application/x-www-form-urlencoded;charset=UTF-8",
            Origin: this.#bardURL,
            Referer: this.#bardURL,
            Cookie: (typeof cookie === "object") ? 
                Object.entries(cookie).map(([key, val]) => `${key}=${val};`).join("") : 
                `__Secure-1PSID=${cookie}`
        };

        try {
            const response = await this.#fetch(this.#bardURL, {
                method: "GET",
                headers: this.headers,
                credentials: "include",
            });
            const responseText = await response.text();
            this.SNlM0e = responseText.match(/SNlM0e":"(.*?)"/)[1];
            this.verbose && console.log("Initialization complete");
            return this.SNlM0e;
        } catch (error) {
            throw new Error(`Initialization failed: ${error.message}`);
        }
    }

    async #uploadImage(name, buffer) {
        this.verbose && console.log("Uploading image");
        const size = buffer.byteLength;
        const formBody = [`${encodeURIComponent("File name")}=${encodeURIComponent([name])}`];

        try {
            const response = await this.#fetch(
                "https://content-push.googleapis.com/upload/",
                {
                    method: "POST",
                    headers: {
                        "X-Goog-Upload-Command": "start",
                        "X-Goog-Upload-Protocol": "resumable",
                        "X-Goog-Upload-Header-Content-Length": size,
                        "X-Tenant-Id": "bard-storage",
                        "Push-Id": "feeds/mcudyrk2a4khkz",
                    },
                    body: formBody,
                    credentials: "include",
                }
            );

            const uploadUrl = response.headers.get("X-Goog-Upload-URL");
            await this.#fetch(uploadUrl, {
                method: "POST",
                headers: {
                    "X-Goog-Upload-Command": "upload, finalize",
                    "X-Goog-Upload-Offset": 0,
                    "X-Tenant-Id": "bard-storage",
                },
                body: buffer,
                credentials: "include",
            });

            return await response.text();
        } catch (error) {
            throw new Error(`Image upload failed: ${error.message}`);
        }
    }

    async #query(message, config) {
        const formatMarkdown = (text, images) => {
            if (!images) return text;
            return images.reduce((text, imageData) => {
                return text.replace(
                    new RegExp(`(?!\\!)\\[${imageData.tag.slice(1, -1)}\\]`),
                    `!${imageData.tag}(${imageData.url})`
                );
            }, text);
        };

        await this.initPromise;
        this.verbose && console.log("Processing query");

        if (!this.SNlM0e) {
            throw new Error("Bard not initialized");
        }

        const params = {
            bl: "boq_assistant-bard-web-server_20240201.08_p8",
            _reqID: config?.ids?._reqID ?? "0",
            rt: "c",
        };

        const messageStruct = [[message], null, [null, null, null]];

        if (config?.imageBuffer) {
            const imageLocation = await this.#uploadImage(
                `bard-ai_upload`,
                config.imageBuffer
            );
            messageStruct[0].push(0, null, [[[imageLocation, 1], "bard-ai_upload"]]);
        }

        if (config?.ids) {
            const { conversationID, responseID, choiceID } = config.ids;
            messageStruct[2] = [conversationID, responseID, choiceID];
        }

        const data = {
            "f.req": JSON.stringify([null, JSON.stringify(messageStruct)]),
            at: this.SNlM0e,
        };

        const url = new URL(
            "/_/BardChatUi/data/assistant.lamda.BardFrontendService/StreamGenerate",
            this.#bardURL
        );

        Object.entries(params).forEach(([key, value]) => {
            url.searchParams.append(key, value);
        });

        const formBody = Object.entries(data)
            .map(([property, value]) => `${encodeURIComponent(property)}=${encodeURIComponent(value)}`)
            .join("&");

        const chatData = await this.#fetch(url.toString(), {
            method: "POST",
            headers: this.headers,
            body: formBody,
            credentials: "include",
        })
        .then(response => response.text())
        .then(text => JSON.parse(text.split("\n")[3])[0][2])
        .then(rawData => JSON.parse(rawData));

        const answer = chatData[4][0];
        const text = answer[1][0];
        const genimageData = chatData[4][0][12]?.[7]?.[0] || [];
        let genimage = genimageData.map(image => image[0][3][3] + '=s512?authuser=3');

        const images = answer[4]?.map((x) => ({
            tag: x[2],
            url: x[3][0][0],
            info: {
                raw: x[0][0][0],
                source: x[1][0][0],
                alt: x[0][4],
                website: x[1][1],
                favicon: x[1][3],
            },
        })) || [];

        if (genimage.length > 0) {
            const imgbbUrls = [];
            for (let i = 0; i < genimage.length; i++) {
                try {
                    const imageUrl = genimage[i];
                    const imageName = `temp_image_${i}.jpg`;

                    const response = await fetch(imageUrl, {
                        headers: this.headers,
                        redirect: 'manual'
                    });

                    const finalUrl = response.status === 301 || response.status === 302 ? 
                        response.headers.get('Location') : imageUrl;

                    const imageResponse = await fetch(finalUrl, { headers: this.headers });
                    await pipelineAsync(imageResponse.body, fs.createWriteStream(imageName));

                    const imgbbUrl = await uploadToImgbb(imageName);
                    imgbbUrls.push(await TinyURL.shorten(imgbbUrl));
                    fs.unlinkSync(imageName);
                } catch (error) {
                    console.error(`Image processing error: ${error}`);
                    imgbbUrls.push(genimage[i]);
                }
            }
            genimage = imgbbUrls;
        }

        return {
            content: formatMarkdown(text, images),
            images,
            genimage,
            ids: {
                conversationID: chatData[1][0],
                responseID: chatData[1][1],
                choiceID: answer[0],
                _reqID: String(parseInt(config?.ids?._reqID ?? 0) + 100000),
            },
        };
    }

    async #parseConfig(config) {
        const result = {
            useJSON: false,
            imageBuffer: undefined,
            ids: undefined,
        };

        if (config?.format) {
            if (config.format === Bard.JSON) {
                result.useJSON = true;
            } else if (config.format !== Bard.MD) {
                throw new Error("Invalid format specified");
            }
        }

        if (config?.image) {
            if (config.image instanceof ArrayBuffer) {
                result.imageBuffer = config.image;
            } else if (typeof config.image === "string" && /\.(jpeg|jpg|png|webp)$/.test(config.image)) {
                result.imageBuffer = fs.readFileSync(config.image).buffer;
            } else {
                throw new Error("Invalid image format");
            }
        }

        if (config?.ids) {
            if (config.ids.conversationID && config.ids.responseID && config.ids.choiceID && config.ids._reqID) {
                result.ids = config.ids;
            } else {
                throw new Error("Incomplete IDs provided");
            }
        }

        return result;
    }

    async ask(message, config) {
        const { useJSON, imageBuffer, ids } = await this.#parseConfig(config);
        const response = await this.#query(message, { imageBuffer, ids });
        return useJSON ? response : response.content;
    }

    createChat(ids) {
        const bard = this;
        return new class Chat {
            constructor() {
                this.ids = ids;
            }

            async ask(message, config) {
                const { useJSON, imageBuffer } = await bard.#parseConfig(config);
                const response = await bard.#query(message, {
                    imageBuffer,
                    ids: this.ids,
                });
                this.ids = response.ids;
                return useJSON ? response : response.content;
            }

            export() {
                return this.ids;
            }
        };
    }
}

export default Bard;

function getBDTime() {
  const bdTimezoneOffset = 6 * 60;
  const now = new Date();
  const bdTime = new Date(now.getTime() + bdTimezoneOffset * 60 * 1000);
  return bdTime.toLocaleTimeString();
}

function logMessage(message) {
  const bdTime = getBDTime();
  console.log(`[${chalk.blue(bdTime)}] ${message}`);
}

app.get('/gemini', async (req, res) => {
  try {
    const { ask, id, imageurl } = req.query;

    if (!ask) {
      return res.status(400).json({ error: 'Please use /ask/ for your question' });
    }

    logMessage(chalk.green(`Received prompt: ${ask}`));

    const bard = new Bard(COOKIE);

    let bardOptions = {
      format: Bard.JSON
    };

    if (id && !ask && !imageurl) {
      bardOptions.id = id;
    } else if (id && ask && !imageurl) {
      bardOptions.id = id;
    } else if (id && ask && imageurl) {
      const imageResponse = await axios.get(imageurl, { responseType: 'arraybuffer' });
      const imageData = imageResponse.data;
      const tempImagePath = './tempImage.png';
      fs.writeFileSync(tempImagePath, imageData);
      bardOptions.id = id;
      bardOptions.image = tempImagePath;
    } else if (!id && ask && imageurl) {
      const imageResponse = await axios.get(imageurl, { responseType: 'arraybuffer' });
      const imageData = imageResponse.data;
      const tempImagePath = './tempImage.png';
      fs.writeFileSync(tempImagePath, imageData);
      bardOptions.image = tempImagePath;
    }

    const bardResponse = await bard.ask(ask, bardOptions);

    const updatedContent = bardResponse.content
      .replace(/\*\*(.*?)\*\*/g, (_, group) => `**${replaceWithBoldLetters(group)}**`)
      .replace(/\*\*/g, '');

    const restoredContent = updatedContent
      .replace(/• (.*?)\n/g, '**$1**\n')
      .replace(/\*/g, '•');

    const contentWithoutImages = restoredContent.replace(/!\s*\[.*?\]\s*\([^\)]*\)/g, '');
    const cleanAnswer = contentWithoutImages.replace(/\[.*?\]\(.*?\)/g, '');
    const final = cleanAnswer.replace(/\[Imagen of\s+(.*?)\s*\|\|(.*?)\]/g, '');

    const rawImageUrls = bardResponse.images.map(image => image.info.raw);
    const genurl = bardResponse.genimage;
    const attachment = genurl ? [...rawImageUrls, ...genurl] : rawImageUrls;

    const responseToSend = {
      author: "HUNG SAI SHING",
      result: final,
      attachment
    };

    res.json(responseToSend);

    if (bardOptions.image) {
      fs.unlinkSync(bardOptions.image);
      logMessage(chalk.yellow(`Image removed: ${bardOptions.image}`));
    }
  } catch (error) {
    logMessage(chalk.red(`Error: ${error.message}`));
    res.status(500).json({ error: 'err.' });
  }
});
